def fetch(ids)
  arr = db.pipelined do
    ids.each { |id| db.hgetall(namespace[id]) }
  end

  res = []

  return res if arr.nil?

  arr.each_with_index do |atts, idx|
    res << model.new(Utils.dict(atts).update(:id => ids[idx]))
  end

  res
end
