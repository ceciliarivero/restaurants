    Replace all the existing elements of a set with a different collection of models. This happens atomically in a MULTI-EXEC block.

def replace(models)
  ids = models.map { |model| model.id }

  key.redis.multi do
    db.del(key)
    ids.each { |id| db.sadd(key, id) }
  end
end
